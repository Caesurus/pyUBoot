bootdelay=3
stdin=serial
stdout=serial
stderr=serial
baudrate=115200
console=ttymxc3
ethaddr=12:34:56:78:90:AB
ethact=FEC
mmcdev=0
mmcpart=1
rootdev=mmcblk2p2
fdtaddr=0x18000000
fdtfile=imx6q-pop-arm2.dtb
loadfdt=fatload mmc ${mmcdev}:${mmcpart} ${fdtaddr} ${fdtfile}
imgaddr=0x12000000
imgfile=zImage
loadimg=fatload mmc ${mmcdev}:${mmcpart} ${imgaddr} ${imgfile}
bootargs=console=${console},${baudrate} root=/dev/${rootdev} rootwait rw
bootcmd=run loadfdt; run loadimg; bootz ${imgaddr} - ${fdtaddr};
